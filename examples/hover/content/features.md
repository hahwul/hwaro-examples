[[extra.features]]
num = "01"
title = "Completion curves"
description = "Every episode gets a play-through curve, not just a download tally, so you can see exactly where listeners lean in or drift."
detail = "Downloads tell you a file moved through a feed. A completion curve tells you what happened after that: the exact second a listener stayed, skipped ahead, or closed the app. We plot it per episode, overlay it against your show's average, and flag the moments that break pattern -- a cold open that loses a fifth of the room, a segment that everyone rewinds. It is the only chart on the dashboard we consider non-negotiable."

[[extra.features]]
num = "02"
title = "One-take publish"
description = "Drop in a WAV or FLAC; Hover masters, transcodes, and pushes it to your feed in one pass, no re-encoding surprises later."
detail = "Upload the highest-resolution file you have. We run a light mastering chain tuned for spoken word -- loudness normalization to -16 LUFS, a gentle de-esser, silence trimming at the head and tail -- then transcode to the bitrate ladder your feed needs. Nothing is re-encoded a second time on our end, so the file a listener downloads in year three sounds identical to the one you approved on day one."

[[extra.features]]
num = "03"
title = "Dynamic ad markers"
description = "Move, swap, or retire a sponsor read after publish without touching the master file or re-uploading the episode."
detail = "Drop a marker at the timestamp where a sponsor read starts and ends. The audio between those markers can be swapped, muted, or replaced entirely after the episode has already gone out, and every listener's next fetch gets the update. No re-upload, no broken chapter marks, no waiting on a CDN cache to expire."

[[extra.features]]
num = "04"
title = "Portable RSS"
description = "Your feed is a plain, open RSS document you can point anywhere. No walled garden, no lock-in, ever."
detail = "We do not run a proprietary player format or require listeners to use an app we built. The feed URL Hover gives you is a standard RSS 2.0 document with the Podcast namespace tags directories expect. Move to another host at any time and take every subscriber, every episode, and every chapter mark with you, unchanged."

[[extra.features]]
num = "05"
title = "Aircheck embeds"
description = "A single embed matches your show's palette and ships with the same per-episode analytics your dashboard sees."
detail = "The embed player is one script tag. It reads your show's accent color and typeface choice from your show settings, so it never looks like a foreign widget bolted onto your site. Every play inside the embed reports back to the same per-episode completion curve you see in your dashboard -- there is no separate, lesser analytics tier for embedded listens."

[[extra.features]]
num = "06"
title = "Supporter loops"
description = "Run an early-access or ad-free feed for paying listeners without routing them through a third-party paywall."
detail = "Supporter loops are a second, private feed generated from the same episode library, with its own release schedule and its own ad-marker rules. Paying listeners get a unique feed URL, revocable at any time from your dashboard. There is no separate app to install and no platform cut on the pledge -- Hover simply manages the feed."

[[extra.features]]
num = "07"
title = "Cold-open scoring"
description = "We flag the first sixty seconds of every episode against your show's historical retention, so you know if the open is working."
detail = "Across thousands of shows we have hosted, the first sixty seconds predicts more about an episode's completion rate than anything that follows. Every new upload gets scored against your own show's baseline the moment it finishes processing, with a plain note -- ahead of pace, on pace, or behind -- so you can decide whether to trim the cold open before it goes out."

The board is small on purpose. Every feature on this page earns its place because a working show asked for it -- there is nothing here chasing a trend, and nothing that requires you to learn a second dashboard to understand your own audience.
